#include "os_main.h"
#include <common.h>

AVD_ERRORCODE os_OutputString( AVDAPP *pApp, char *sOutputString )
{
    AVD_ERRORCODE Results = AVDERR_NOERROR;

    if ( AVD_NULL != sOutputString ) puts(sOutputString);

    return( (AVD_ERRORCODE)Results );
}
