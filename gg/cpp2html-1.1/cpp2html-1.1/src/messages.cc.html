// messages.cc

#include "messages.h"

static Messages *_messager = 0 ;

void printMessage( char *s, ostream &stream ) {
  if ( _messager )
    _messager->printMessage(s,stream) ;
}

void printWarning( char *s, ostream &stream ) {
  if ( _messager )
    _messager->printWarning(s,stream) ;
}

void printError( char *s, ostream &stream ) {
  if ( _messager )
    _messager->printError(s,stream) ;
}

void setMessager( Messages *m ) {
  if ( m )
    _messager = m ;
}
