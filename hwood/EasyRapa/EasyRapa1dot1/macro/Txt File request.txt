Function macroname()
	;request a txt file to be included
	;Note macroname is mandatory and will be automatically replaced by EasyRapa with the name reflecting the itm ID and its functionality

		f$=FileRequest("Select a file",{Path=""   , Filters="txt"})
		If f$=""
			moai.Request("File request", "File selection was aborted", "OK")
		EndIf
		Return(f$)
EndFunction
