Function macroname(ID$,title, dimw, dimh)
	; script to create a simple dialog 
	;Note macroname is mandatory and will be automatically replaced by EasyRapa with the name reflecting the itm ID and its functionality

	if ID$=Nil then ID$="MyDialog"
	if title=Nil then title="Title"
	if dimw=Nil then dimw=200
	if dimh=Nil then dimh=200
	
	local dialogxml$=[[<?xml version="1.0" encoding="iso-8859-1"?>

	<dialog ID="]]..id$..[[" title="]]..title..[[" width="]]..dimw..[["height="]]..dimh..[[" >
		<hgroup>
		<text> This is a simple dialog  </text>
		</hgroup>
	</dialog>

	moai.CreateDialog (dialogxml$)

	moai.DoMethod("]]..id$..[[", "showmodal")	
	]]	
EndFunction
