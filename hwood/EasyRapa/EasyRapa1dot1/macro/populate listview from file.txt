Function macroname(file$,delwith$)
	;file$ = file name to read
	;delwith$ = filelds delimited by character
	;Note macroname is mandatory and will be automatically replaced by EasyRapa with the name reflecting the itm ID and its functionality
	;THIS MACRO IS UNCOMPLETED AND IF USED WILL GIVE ERROR ON TESTMODE

	if delwith$=Nil then delwith$="|"

	s$, len = FileToString(file$)
	lines,tl=SplitStr(s$,Chr(10))
	
	For k=0 To tl-1 
		If lines[k]<>""
			Lines[k]=ReplaceStr(lines[k],Chr(34),"")
;			DebugPrint(lines[k]) 
			s$,noffields=SplitStr(lines[k],delwith$)
			moai.DoMethod("##ID##", "Insert", "Bottom" ,s$[0],s$[1],s$[2],s$[3],s$[4],s$[5],s$[6]) 
		EndIf
	Next	
EndFunction
