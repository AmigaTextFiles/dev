                        ======= START.DATA =======

Pattern:
	DATAFILE;PATTERN=nr,start.l,length;
	NORMFILE;PATTERN=nr,plik$;
	Jeûeli dajemy pattern w chunky, to dadajemy do nr+127

Obrazki tytuîowe:
	DATAFILE;DEMO_PIC=nr,start,length;
	NORMFILE;DEMO_PIC=nr,plik$;
	Jeûeli dajemy obrazek w chunky, to dodajemy do nr+10

Panel:
        Jest to obrazek tytuîowy nr 21 (planar) i nr 31 (chunky)
	Obrazek w trybie low (jako anim5 lub anim7) jest pod numerem 41

Plik link w menu:
        NORMFILE;LINK_FILE=LINK;
	Otworzy plik TEKSTY/LINK.txt
        DATAFILE;LINK_FILE=START,LENGTH;
	Zaladuje z DATA.FILE

Plik menu:
	NORMFILE;MENU=0,MENU.TXT;     (zaîaduje z TEKSTY/DATA/MENU.txt)
	DATAFILE;MENU=0,START,LENGTH;

Moduî muzyczny:
	NORMFILE;MODULE=nr,FILENAME,Nazwa_w_requesterze;
	DATAFILE;MODULE=nr,START,LENGTH,Nazwa_w_requesterze;

Strefa:
	ZONE;nr=x,t,w,h;

Gadget:
	Planar mode
	NORMFILE;GADGET=nr,frames,filename.iff;
	DATAFILE;GADGET=nr,frames,start,length;

	Chunky mode
	NORMFILE;GADGET=nr+10,frames,filename.chk;
	DATAFILE;GADGET=nr+10,frames,start,length;

Iloôê obrazków tytuîowych:
	DEMO_PICS;nr;
