WBStartup
;NoCli

;ConfigDev Structure
DEFTYPE.ConfigDev *Config
*Config=FindConfigDev_(0,-1,-1)

ManufSTR.l=AllocMem_(50,#MEMF_CLEAR)
ProdSTR.l=AllocMem_(50,#MEMF_CLEAR)



Dim Tags.TagItem(4)

Tags(0)\ti_Tag=#IDTAG_ConfigDev,*Config
Tags(1)\ti_Tag=#IDTAG_ManufStr,ManufSTR
Tags(2)\ti_Tag=#IDTAG_ProdStr,ProdSTR
Tags(3)\ti_Tag=#IDTAG_StrLength,50
Tags(4)\ti_Tag=#TAG_DONE

NPrint ""
NPrint "Expansion - (c)Luca Ferraris"
NPrint "- List of Zorro I/II/III boards found in your system -"
NPrint ""
While *Config
  Error=IdExpansion_(&Tags(0))
  If Error=0
    NPrint "Producer: ",Peek$(ManufSTR)
    NPrint "Product:  ",Peek$(ProdSTR)
    *Config=*Config\cd_NextCD
    Tags(0)\ti_Tag=#IDTAG_ConfigDev,*Config
  End If
Wend
Delay_ 50
