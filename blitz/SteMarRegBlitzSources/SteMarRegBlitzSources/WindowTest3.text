WBStartup
WbToScreen 0
XLenW.w=200:YLenW.w=150:XPosW.w=(WBWidth-XLenW)/2:YPosW.w=(WBHeight-YLenW)/2
Window 0,XPosW,YPosW,XLenW,YLenW,$1008,"(De)select me!",-1,-1
ProgramEnd.b=False
Repeat
 WindowEvent.l=WaitEvent
 Select WindowEvent
 Case $200
  ProgramEnd=True
 Case $40000
  WSize 200,150
 Case $80000
  WSize 200,11
 End Select
Until ProgramEnd=True
End
