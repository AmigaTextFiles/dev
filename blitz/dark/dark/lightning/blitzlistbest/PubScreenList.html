Statement GetPubScreenList {}

  ad.l=LockPubScreenList_
  UnlockPubScreenList_

  ad=Peek.l(ad)

  maxnum.w=50
  Repeat
    If Peek.l(ad+10)<>0
      sname$=Peek$(Peek.l(ad+10))
      If LockPubScreen_(sname$)
        NPrint sname$
        UnlockPubScreen_ sname$,0
      EndIf
       ad=Peek.l(ad)
       maxnum-1
    Else
      Statement Return
    EndIf
  Until maxnum=0
End Statement
