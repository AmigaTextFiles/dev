Typen umwandeln

 

    L = long Register, D = double Register
    S = string Variable


    val_l       S, L;               String nach Zahl
    val_d       S, D;

    str_l       L, S;               Zahl nach string
    str_d       D, S;

    2int        D, L;               double (Fliesskomma) nach int (Ganzzahl)
    2double     L, D;               int nach double


    char        L, S;               ASCII-Code nach String: 65 -> A
    asc         S, L;               String nach ASCII-Code: A  -> 65
