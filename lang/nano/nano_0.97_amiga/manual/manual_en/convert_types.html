Convert Types

 

    L = long register, D = double register
    S = string variable


    val_l       S, L;               string to number
    val_d       S, D;

    str_l       L, S;               number to string
    str_d       D, S;

    2int        D, L;               double to int
    2double     L, D;               int to double


    char        L, S;               ASCII-code to string: 65 -> A
    asc         S, L;               string to ASCII-code: A  -> 65
