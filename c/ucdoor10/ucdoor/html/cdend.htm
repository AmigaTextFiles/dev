void cd_End(int errcod);
   
   ---
      
   Purpose: exit the door immediately, use in case of error.
   
   Results: sets 'udc_result' to 'errcode' 
   
   Usage  :     if(cd_CarrierLost())
                {
                    cd_End(ERROR_CARRIER);
                }
                
                or
                
                if(xxxx==9)
                {
                    cd_End(ERROR_ABORTED);
                }
